285template <typename T, typename MatTraits, typename OutTraits, typename BackendTag>
286void matrix_power_into(TensorView<T const, MatTraits> mat, unsigned int power, TensorView<T, OutTraits> out,
287 BackendTag tag)
288{
289 detail::require_square(mat);
290 detail::require_square(out);
291 if (mat.rows() != out.rows() || mat.cols() != out.cols())
292 {
293 throw std::invalid_argument("output matrix has incompatible dimensions for matrix_power");
294 }
295
296 using value_type = detail::value_type_t<decltype(mat)>;
297 using tensor_type = uni20::BasicTensor<value_type, stdex::dextents<index_type, 2>, VectorStorage>;
298
299 tensor_type result(detail::make_extents(mat));
300 ::uni20::linalg::fill_identity(result.view(), tag);
301
302 if (power == 0U)
303 {
304 ::uni20::linalg::copy(result.view(), out, tag);
305 return;
306 }
307
308 tensor_type base = detail::clone(mat, tag);
309 tensor_type scratch(detail::make_extents(mat));
310
311 unsigned int exponent = power;
312 while (exponent > 0U)
313 {
314 if ((exponent & 1U) != 0U)
315 {
316 ::uni20::linalg::multiply_into(result.view(), base.view(), scratch.view(), tag);
317 ::uni20::linalg::copy(scratch.view(), result.view(), tag);
318 }
319 exponent >>= 1U;
320 if (exponent != 0U)
321 {
322 ::uni20::linalg::multiply_into(base.view(), base.view(), scratch.view(), tag);
323 ::uni20::linalg::copy(scratch.view(), base.view(), tag);
324 }
325 }
326
327 ::uni20::linalg::copy(result.view(), out, tag);
328}
