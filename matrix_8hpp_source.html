174template <typename T> Matrix<T> matrix_power(Matrix<T> const& mat, unsigned int power)
175{
176 if (mat.rows() != mat.cols())
177 {
178 throw std::invalid_argument("matrix_power requires a square matrix");
179 }
180
181 if (power == 0U)
182 {
183 return make_identity<T>(mat.rows());
184 }
185
186 Matrix<T> result = make_identity<T>(mat.rows());
187 Matrix<T> base = mat;
188 unsigned int exponent = power;
189 while (exponent > 0U)
190 {
191 if ((exponent & 1U) != 0U)
192 {
193 result = multiply(result, base);
194 }
195 exponent >>= 1U;
196 if (exponent != 0U)
197 {
198 base = multiply(base, base);
199 }
200 }
201
202 return result;
203}

239template <typename T> Matrix<T> solve_linear_system(Matrix<T> A, Matrix<T> B)
240{
241 if (A.rows() != A.cols() || A.rows() != B.rows())
242 {
243 throw std::invalid_argument("solve_linear_system requires square coefficient matrix");
244 }
245
246 std::size_t n = A.rows();
247 std::size_t nrhs = B.cols();
248
249 for (std::size_t k = 0; k < n; ++k)
250 {
251 std::size_t pivot_row = k;
252 double pivot_value = std::abs(A(k, k));
253 for (std::size_t i = k + 1; i < n; ++i)
254 {
255 double candidate = std::abs(A(i, k));
256 if (candidate > pivot_value)
257 {
258 pivot_value = candidate;
259 pivot_row = i;
260 }
261 }
262
263 if (pivot_value == 0.0)
264 {
265 throw std::runtime_error("singular matrix in solve_linear_system");
266 }
267
268 swap_rows(A, k, pivot_row);
269 swap_rows(B, k, pivot_row);
270
271 T const pivot = A(k, k);
272 for (std::size_t i = k + 1; i < n; ++i)
273 {
274 T const factor = A(i, k) / pivot;
275 if (factor == T{})
276 {
277 continue;
278 }
279 A(i, k) = T{};
280 for (std::size_t j = k + 1; j < n; ++j)
281 {
282 A(i, j) -= factor * A(k, j);
283 }
284 for (std::size_t j = 0; j < nrhs; ++j)
285 {
286 B(i, j) -= factor * B(k, j);
287 }
288 }
289 }
290
291 Matrix<T> X = B;
292 for (int i = static_cast<int>(n) - 1; i >= 0; --i)
293 {
294 T const pivot = A(static_cast<std::size_t>(i), static_cast<std::size_t>(i));
295 for (std::size_t j = 0; j < nrhs; ++j)
296 {
297 T value = X(static_cast<std::size_t>(i), j);
298 for (std::size_t k = static_cast<std::size_t>(i) + 1; k < n; ++k)
299 {
300 value -= A(static_cast<std::size_t>(i), k) * X(k, j);
301 }
302 X(static_cast<std::size_t>(i), j) = value / pivot;
303 }
304 }
