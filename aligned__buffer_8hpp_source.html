78inline void* allocate_raw(std::size_t bytes, std::size_t align)
79{
80 // if the buffer is smaller than the requested alignment,
81 // drop the alignment to avoid wasted space:
82 if (bytes < align)
83 {
84 // bit_floor(bytes) is the largest power-of-two ≤ bytes
85 auto pf = std::bit_floor(bytes);
86 // never go below pointer alignment:
87 align = std::max<std::size_t>(pf, alignof(void*));
88 }
89
90 void* ptr = nullptr;
91#if defined(_MSC_VER)
92 ptr = _aligned_malloc(bytes, align);
93 if (!ptr) throw std::bad_alloc();
94#else
95 // posix_memalign only cares that:
96 // - align is pow-2 and ≥ sizeof(void*)
97 // - size can be any value
98 if (int err = posix_memalign(&ptr, align, bytes); err)
99 {
100 throw std::bad_alloc();
101 }
102#endif
103
104 return ptr;
105}

151template <typename T> auto allocate_temporary_buffer(std::size_t N, std::size_t align = 64)
152{
153 if constexpr (uninitialized_ok<T>)
154 {
155 // raw memory, no ctors/dtors
156 return allocate_uninitialized_buffer<T>(N, align);
157 }
158 else
159 {
160 // allocate raw
161 void* raw = detail::allocate_raw(sizeof(T) * N, align);
162 T* ptr = static_cast<T*>(raw);
163
164 // default-construct each element

168 }
