22{
23 static_assert(std::numeric_limits<T>::is_iec559, "float_distance requires IEEE-754 floating point");
24
25 using UInt =
26 std::conditional_t<sizeof(T) == 4, std::uint32_t, std::conditional_t<sizeof(T) == 8, std::uint64_t, __uint128_t>>;
27
28 if (std::isnan(a) || std::isnan(b))
29 {
30 return std::numeric_limits<long long>::max();
31 }
32 if (std::isinf(a) || std::isinf(b))

49}
