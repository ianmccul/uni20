38template <IsAsyncTaskPromise T> BasicAsyncTask<T> BasicAsyncTask<T>::make_sole_owner(BasicAsyncTask<T>&& task)

72template <IsAsyncTaskPromise T> BasicAsyncTask<T>::handle_type BasicAsyncTask<T>::release_handle()
73{
74 TRACE_MODULE(ASYNC, "BasicAsyncTask::release_handle", h_);
75 CHECK(h_);
76 if (!h_.promise().release_awaiter()) PANIC("Attempt to resume() a non-exclusive AsyncTask");
77
78 bool to_destroy = h_.promise().is_cancel_on_resume();
79
80 auto handle = h_;
81 h_ = nullptr; // Always drop ownership, we are now effectively in a 'moved from' state
82 // we need to do this *before* calling h_.resume(), because it is possible that this AsyncTask
83 // is on that coroutine frame, and it might get destroyed.
84
85 if (to_destroy)
86 {
87 CHECK(this->can_destroy_coroutine(handle), "unexpected destruction of an active AsyncTask without cancellation",
88 handle);
89 // if we need to destroy the coroutine, recursively destroy any continuations as well
90 while (handle)
91 {
92 TRACE_MODULE(ASYNC, "Destroying AsyncTask due to cancellation", handle);
93 handle = handle.promise().destroy_with_continuation();
94 }
95 }
96 else
97 {
98 handle.promise().mark_started();
99 }
100 return handle;
101}
