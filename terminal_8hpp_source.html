29inline std::string quote_shell(std::string_view s)
30{
31 // Characters that require quoting/escaping
32 constexpr std::string_view special_chars = " \t()[]*\\\"";
33
34 // Check if any character in s is one of the special ones
35 bool needs_quote = std::ranges::any_of(s, [&](char c) { return special_chars.find(c) != std::string_view::npos; });
36
37 if (needs_quote)
38 {
39 std::string result;
40 // Reserve extra space (worst-case: every character may need escaping)
41 result.reserve(s.size() * 2);
42 for (char c : s)
43 {
44 if (c == '\\')
45 {
46 result.append("\\\\");
47 }
48 else if (c == '"')
49 {
50 result.append("\\\"");
51 }
52 else
53 {
54 result.push_back(c);
55 }
56 }
57 return "\"" + result + "\"";
58 }
59 return std::string(s);
60}

287// parseTerminalStyle expects a style string made up of one or more style components,
288// separated by commas. Each component can optionally specify a target using "fg:" for
289// foreground or "bg:" for background (default is foreground), followed by a color spec,
290// and then optional attributes separated by semicolons.
291//
292// Color specifications can be provided as:
293// - A named color (e.g., "Red", "Blue", "lightgray", etc.)
294// - An RGB color in function notation (e.g., "rgb(255,0,0)")
295// - A hexadecimal color (e.g., "#FF0000" or "#F00")
296//
297// Any tokens following the color (separated by semicolons) are treated as text attributes,
298// such as "Bold", "Dim", or "Underline".
299//
300// Examples:
301// "Red;Bold" -> Sets foreground to Red with Bold.
302// "fg:rgb(255,0,0);Bold" -> Sets foreground to rgb(255,0,0) with Bold.
303// "bg:#00FF00;Dim" -> Sets background to green (#00FF00) with Dim.
304// "fg:LightGray;Underline, bg:darkgray"
305// -> Sets foreground to LightGray with Underline,
306// and background to darkgray.
