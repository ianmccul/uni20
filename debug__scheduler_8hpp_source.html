186inline void DebugScheduler::run()
187{
188 DEBUG_TRACE_MODULE(ASYNC, "DebugScheduler::run");
189 if (Blocked_)
190 {
191 DEBUG_TRACE_MODULE(ASYNC, "run() on a blocked DebugQueue: doing nothing");
192 return;
193 }
194 TRACE_MODULE(ASYNC, "Got some coroutines to resume", Handles_.size());
195
196 std::vector<AsyncTask> H;
197 std::swap(H, Handles_);
198 std::reverse(H.begin(), H.end());
199 for (auto&& h : H)
200 {
201 TRACE_MODULE(ASYNC, "resuming coroutine...", &h, h.h_);
202 h.resume();
203 CHECK(!h);
204 TRACE_MODULE(ASYNC, "here", &h, Handles_.size());
205 }
206}
