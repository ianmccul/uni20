81// =============================================================================================
82// EpochContext Invariants
83// =============================================================================================
84//
85// Exactly one of the following must hold at all times:
86//
87// * phase_ == Pending
88//
89// * phase_ == Started
90// && total_writers_ == 0
91//
92// * phase_ == Writing
93// && num_writers_ > 0
94//
95// * phase_ == Reading
96// && num_writers_ == 0
97// && (num_readers_ > 0 || next_epoch_ == nullptr)
98//
99// * phase_ == Finished
100//
101// Normally, num_writers_ <= 1. (Not sure if there is a use for > 1, as a transient?)
102//
103// ---------------------------------------------------------------------------------------------
104//
105// Destruction invariants:
106//
107// * Destruction while (num_writers_ > 0 || num_readers_ > 0)
108// is erroneous (assertion failure).
109//
110// ---------------------------------------------------------------------------------------------
111//
112// Phase transition rules:
113//
114// * Pending → Started
115// Triggered by start_epoch()
116//
117// * Started → Writing
118// Triggered by total_writers_ > 0.
119//
120// * Writing → Reading
121// Triggered when num_writers_ drops to to 0.
122// (i.e. writer has finished).
123// \pre num_writers_ = 0
124//
125// * Reading → Finished
126// Triggered when num_readers_ drops to 0 && next_epoch_ != nullptr
127// or next_epoch_ is set and num_readers == 0
128// Invokes next_epoch_->start() and passes on cancelled_ and eptr_
129// \pre num_readers_ == 0 && next_epoch_ != nullptr
130//
131// * Writing → Finished
132// Triggered when num_writers_ drops to to 0 && num_readers_ == 0 && next_eopch_ != nullptr
133// (i.e. writer has finished, and there are no readers).
134// \pre num_writers_ = 0 && num_readers_ == 0 && next_epoch_ != nullptr
135//
136// Note that immediate transitions are possible.
137//
138// ---------------------------------------------------------------------------------------------
139//
140// Registration rules:
141//
142// * Writers can only be added when phase_ == Pending || phase_ == Started.
143// * Readers can only be added when phase_ != Finished.
144//
145// =============================================================================================
146//
147// Cancellation:
148// An Epoch chain can be cancelled at any time. This indicates that the final result of this
149// calculation is not going to be used and Reader tasks will either need to handle the cancel()
150// or they will get an exception.
151//
152// Initialization status is handled separately. The EpochContext simply handles writers and readers.
153
154// WriteBuffer participates in epoch scheduling only; value construction state is tracked by shared_storage<T>.
155// If a writer is released without emplacing/writing a value, the storage can remain unconstructed for this epoch.
156//
157// If you co_await on a ReadBuffer that points to an EpochContext that is in the cancelled state, then
158// it will either throw an exception, or cancel the coroutine (if .or_cancel() modifier was used).
159//
160// If there is a next_epoch_, then the cancellation and exception status is propogated forwards.
161//
162// co_await on a WriteBuffer on a cancelled or exception EpochContext passes the error into the coroutine.
163//

547 void advance_writing_locked(std::unique_lock<std::mutex> lock) noexcept
548 {
549 DEBUG_TRACE_MODULE(ASYNC, "EpochContext::advance_writing_locked", this, counter_);
550 DEBUG_PRECONDITION(phase_ == Phase::Started);
551 phase_ = Phase::Writing;

556 writer_tasks_.pop_back();
557 this->execute_writer_locked(std::move(lock), std::move(NextTask));

565 {
566 DEBUG_TRACE_MODULE(ASYNC, "EpochContext::advance_reading_locked", this, counter_);
567 DEBUG_PRECONDITION(phase_ == Phase::Writing);
568 phase_ = Phase::Reading;
569
570 if (num_readers_ > 0)
571 {
572 this->execute_readers_locked(std::move(lock));

575
576 if (num_readers_ == 0 && next_epoch_ != nullptr) this->advance_finished_locked(std::move(lock));

591 void execute_writer_locked(std::unique_lock<std::mutex> lock, TaskState task)
592 {
593 DEBUG_TRACE_MODULE(ASYNC, "EpochContext::execute_writer_locked", this, counter_);
594 DEBUG_CHECK(phase_ == Phase::Writing);
595 DEBUG_CHECK(!writer_active_);
596
597 if (eptr_)
598 {
599 if (task.cancel_on_exception)
600 task.task.set_cancel_on_resume();
601 else
602 task.task.exception_on_resume(eptr_);
603 }
604 writer_active_ = true;
605 lock.unlock();
606 AsyncTask::reschedule(std::move(task.task));
607 }
608
609 void execute_readers_locked(std::unique_lock<std::mutex> lock)
610 {
611 DEBUG_TRACE_MODULE(ASYNC, "EpochContext::execute_readers_locked", this, counter_);
612
613 // When we reschedule (or cancel) a task, we need to assume that the coroutine frame may get
614 // destroyed at any time, which might result in the last reference to this EpochContext itself
615 // disappearing. This means that we cannot reference *this at all once we start rescheduling
616 // tasks. Hence copy all of the data structures we need into local storage and release the lock.
617 std::exception_ptr my_eptr = eptr_;
618 std::vector<TaskState> my_reader_tasks;
619 std::swap(reader_tasks_, my_reader_tasks);
620 lock.unlock();
621
622 for (auto&& task : my_reader_tasks)
623 {
624 if (my_eptr)
625 {
626 if (task.cancel_on_exception)
627 task.task.set_cancel_on_resume();
628 else
629 task.task.exception_on_resume(my_eptr);
630 }
631 AsyncTask::reschedule(std::move(task.task));
632 }
