146template <typename Awt> struct TryAwaiter
147{
148 using Awaiter = std::remove_reference_t<Awt>;
149 using StoredAwaiter = std::conditional_t<std::is_lvalue_reference_v<Awt>, Awaiter*, Awt>;

152
153 constexpr explicit TryAwaiter(Awt aw) noexcept : awaiter_(store(std::forward<Awt>(aw))) {}
154
155 bool await_ready() const noexcept { return true; }

157 void await_suspend(AsyncTask t) noexcept { access().await_suspend(std::move(t)); }
158
159 auto await_resume() noexcept
160 {
161 auto& inner = access();
162 using R = decltype(inner.await_resume());
163 using StoreT =
164 std::conditional_t<std::is_lvalue_reference_v<R>, std::reference_wrapper<std::remove_reference_t<R>>, R>;
165
166 if (!inner.await_ready())
167 {
168 return std::optional<StoreT>{};
169 }
170
171 if constexpr (std::is_lvalue_reference_v<R>)
172 {
173 return std::optional<StoreT>{std::ref(inner.await_resume())};
174 }
175 else
176 {
177 return std::optional<StoreT>{inner.await_resume()};
178 }
179 }
