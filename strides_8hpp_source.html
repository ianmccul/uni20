99template <std::size_t N, std::size_t R> void sort_and_merge_left(static_vector<extent_strides<N>, R>& out)
100{
101 if (out.size() <= 1) return;
102
103 std::sort(out.begin(), out.end(),
104 [](auto const& lhs, auto const& rhs) { return std::abs(lhs.strides[0]) < std::abs(rhs.strides[0]); });
105
106 extent_strides<N> current = out[0];
107 std::size_t write = 0;
108 for (std::size_t i = 1; i < out.size(); ++i)
109 {
110 auto candidate = out[i];
111 if (candidate.can_merge_with_inner(current))
112 {
113 candidate.merge_with_inner(current);
114 current = candidate;
115 }
116 else
117 {
118 out[write++] = current;
119 current = candidate;
120 }
121 }
122
123 out[write++] = current;
124 out.resize(write);
125}

233template <StridedMdspan AType, StridedMdspan BType, StridedMdspan CType, std::size_t N>
234auto extract_strides(AType const& A, BType const& B,
235 std::array<std::pair<std::size_t, std::size_t>, N> const& contractDims, CType const& C)
236{
237 constexpr std::size_t MR = AType::rank() - N; // rank of the M group (A/C legs that are not contracted over)
238 constexpr std::size_t NR = BType::rank() - N; // rank of the N group (B/C legs that are not contracted over)
239 constexpr std::size_t KR = N; // rank of the K group (A/Blegs that are contracted over)
240
241 static_vector<extent_strides<2>, MR> Mgroup;
242 static_vector<extent_strides<2>, NR> Ngroup;
243 static_vector<extent_strides<2>, KR> Kgroup;
244
245 // Assemble the K array of the contracted legs and mark which legs of A and B are contracted over
246 std::array<bool, AType::rank()> AContracted{false};
247 std::array<bool, BType::rank()> BContracted{false};
248 for (std::size_t i = 0; i < N; ++i)
249 {
250 // check that the extents match
251 auto [ai, bi] = contractDims[i];
252 ERROR_IF(A.extent(ai) != B.extent(bi), "Extent along tensor contraction dimension does not match", ai, bi);
253 AContracted[ai] = true;
254 BContracted[bi] = true;
255 Kgroup.emplace_back(A.extent(ai), A.stride(ai), B.stride(bi));
256 }
257 // Now fill out the uncontracted dimensions and verify that they match with C
258 std::size_t ci = 0;
259 for (std::size_t ai = 0; ai < AType::rank(); ++ai)
260 {
261 if (!AContracted[ai])
262 {
263 ERROR_IF(A.extent(ai) != C.extent(ci), "Extent along uncontracted dimension does not match", ai, ci);
264 Mgroup.emplace_back(A.extent(ai), A.stride(ai), C.stride(ci));
265 ++ci;
266 }
267 }
268 for (std::size_t bi = 0; bi < BType::rank(); ++bi)
269 {
270 if (!BContracted[bi])
271 {
272 ERROR_IF(B.extent(bi) != C.extent(ci), "Extent along uncontracted dimension does not match", bi, ci);
273 Ngroup.emplace_back(B.extent(bi), B.stride(bi), C.stride(ci));
274 ++ci;
275 }
276 }
277
278 merge_strides_right(Mgroup);
279 merge_strides_right(Ngroup);
280 merge_strides_right(Kgroup);
281
282 return std::tuple{Mgroup, Ngroup, Kgroup};
283}
