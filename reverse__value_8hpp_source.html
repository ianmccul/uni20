18template <typename T, typename U> AsyncTask async_accumulate(ReadBuffer<T> a_, ReadBuffer<U> b_, WriteBuffer<T> out_)
19{
20 auto a = co_await a_.transfer().maybe();
21 if (a)
22 {
23 T value = std::move(*a).get_release();
24 auto b = co_await b_.transfer().maybe();
25 if (b) value += std::move(*b).get_release();
26 co_await out_ = std::move(value);
27 }
28 else
29 {

32 // U value = (co_await b_.transfer().or_cancel()).get();
33 auto b = co_await b_.transfer().or_cancel();
34 U value = b.get();
35 b.release();
36 co_await out_ += std::move(value);
37 }
