22template <typename T> AsyncTask co_sin(ReadBuffer<T> in, WriteBuffer<T> out)
23{
24 using std::sin;
25 // The owning proxy from `co_await in.transfer()` transfers ownership of the read buffer.
26 // We explicitly call `release()` immediately after `get()`, before write-buffer acquisition.
27 // This ordering matters because it is possible that
28 // there are data dependencies that mean that we cannot get the WriteBuffer until after the Read
29 // has finished. For example consider
30 // Async<double> x = 1.0;
31 // auto rbuf = x.read();
32 // x *= 2;
33 // auto wbuf = x.write();
34 // schedule(co_sin(std::move(rbuf), std::move(wbuf)));
35 // GCC 13 limitation: `(co_await in.transfer()).get()` can fail with
36 // "insufficient contextual information to determine type" in templates.
37 // This is fixed in GCC 14+, where the one-liner below is preferred:
38 // auto const input = (co_await in.transfer()).get();
39 auto input_buffer = co_await in.transfer();
40 auto const input = input_buffer.get();
41 input_buffer.release();
42 co_await out = sin(input);
43}
