151 template <typename MDS, typename... Fixed>
152 auto
153 smart_submdspan(const MDS& mds, Fixed... fixed)
154{
155 constexpr std::size_t total_rank = MDS::rank();
156 constexpr std::size_t num_fixed = sizeof...(fixed);
157 static_assert(num_fixed <= total_rank);
158
159 auto rest = []<std::size_t... Is>(std::index_sequence<Is...>)
160 {
161 return std::make_tuple((void(Is), stdex::full_extent)...);
162 }
163 (std::make_index_sequence<total_rank - num_fixed>{});
164
165 return std::apply([&](auto... rest_extents) { return stdex::submdspan(mds, fixed..., rest_extents...); }, rest);
166}
