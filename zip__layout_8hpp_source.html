27// Helper function to merge extents. If the extents are different, then one of them must be dynamic,
28// otherwise they are not compatible. If one is dynamic and one is static, then return the static extent.
29
30template <std::size_t A, std::size_t B> static constexpr std::size_t merge_extent()
31{
32 if constexpr (A == B)
33 return A;
34 else if constexpr (A == stdex::dynamic_extent)
35 return B;
36 else if constexpr (B == stdex::dynamic_extent)
37 return A;
38 else
39 static_assert(A == B || A == stdex::dynamic_extent || B == stdex::dynamic_extent, "Incompatible static extents");
40 return 0;
41}
42
43// Primary template: merge_pack over a pack of N extents
44template <std::size_t... Ns> struct merge_pack;
45
46// Base case: single value
47template <std::size_t A> struct merge_pack<A>
48{
49 static constexpr std::size_t value = A;
50};
51
52// Recursive case: merge first two, then fold in the rest
53template <std::size_t A, std::size_t B, std::size_t... Rest> struct merge_pack<A, B, Rest...>
54{
55 static constexpr std::size_t value = merge_pack<merge_extent<A, B>(), Rest...>::value;
56};

58// Given a set of mdspans, determine the common extents using the merge_extent() function above
59template <typename FirstSpan, typename... OtherSpans> struct common_extents
60{
61 static constexpr std::size_t NumSpans = 1 + sizeof...(OtherSpans);
62
63 // 1) All spans must have the same rank R
64 static constexpr std::size_t R = FirstSpan::rank();
65 static_assert((... && (OtherSpans::rank() == R)), "common_extents: ranks must match");
66
67 using index_type = FirstSpan::index_type;
68
69 private:
70 // Fold-merge the static extents at dimension I
71 template <std::size_t I> static constexpr std::size_t merged_static_extent()
72 {
73 // build a pack of the static-extent<I> of each Span
74 return merge_pack<FirstSpan::extents_type::static_extent(I),
75 OtherSpans::extents_type::static_extent(I)...>::value;
76 }
77
78 // Build the extents type by expanding merged_static_extent<Is>()...
79 template <std::size_t... Is> static constexpr auto make_extents(std::index_sequence<Is...>)
80 {
81 return stdex::extents<index_type,
82 merged_static_extent<Is>()... // one per dimension
83 >{};
84 }
85
86 public:
88 using type = decltype(make_extents(std::make_index_sequence<R>{}));
89
90 private:
91 template <std::size_t... Is>
92
93 static type make_impl(std::index_sequence<Is...>, FirstSpan const& first, OtherSpans const&... otherspans)
94 {
95 // Construct the extents object. All of the extents of each span must agree.
96 type ext{first.template extent(Is)...};
97
98 auto do_check = [&](auto const& sp) { CHECK_EQUAL(ext, sp.extents()); };
99
100 (do_check(otherspans), ...);
101
102 return ext;
103 }
104
105 public:
107 static type make(FirstSpan const& firstspan, OtherSpans const&... otherspans)
108 {
109 return make_impl(std::make_index_sequence<R>{}, firstspan, otherspans...);
110 }
111};

359 template <typename... Idx> constexpr offset_type operator()(Idx... idxs) const noexcept
360 {
361 static_assert(sizeof...(Idx) == Ext::rank(), "Wrong number of indices");
362 // pack the multi-index into an array
363 std::array<index_type, Ext::rank()> ix{static_cast<index_type>(idxs)...};
364 // dispatch over spans

366 }

369 // For a given span S, compute its offset by folding over all dims D:
370 template <std::size_t S, std::size_t... D>
371 constexpr index_type span_offset(std::array<index_type, Ext::rank()> const& ix,
372 std::index_sequence<D...>) const noexcept
373 {
374 // fold: 0 + (stride[S][0]*ix[0]) + (stride[S][1]*ix[1]) + …
375 return ((all_strides_[S][D] * ix[D]) + ... + index_type(0));
376 }

379 template <std::size_t... S>
380 constexpr offset_type offset_impl(std::array<index_type, Ext::rank()> const& ix,
381 std::index_sequence<S...>) const noexcept
382 {
383 // pack-expand: span_offset<0>(ix), span_offset<1>(ix), …
384 return offset_type{span_offset<S>(ix, std::make_index_sequence<Ext::rank()>{})...};
385 }
