223 void register_exception_sink(ExceptionSinkNode& node, std::shared_ptr<EpochContext> epoch, bool explicit_sink)
224 {

240 void unregister_exception_sink(ExceptionSinkNode& node, bool from_destructor) noexcept
241 {
242 if (node.owner != this) return;
243 if (from_destructor && node.explicit_sink && std::uncaught_exceptions() > 0)
244 {
245 CHECK(
246 false,
247 "propagate_exceptions_to sink destroyed during exception unwinding before coroutine unhandled_exception()");
248 }
249
250 if (node.prev)
251 node.prev->next = node.next;
252 else
253 exception_sinks_head_ = node.next;
254
255 if (node.next) node.next->prev = node.prev;

257 node.owner = nullptr;
258 node.prev = nullptr;
259 node.next = nullptr;
260 node.epoch.reset();
261 node.explicit_sink = false;

373 static std::coroutine_handle<> resolve_await_suspend_result(std::coroutine_handle<promise_type> h, AsyncTask&& t)
374 {
375 // null handle means suspend the current coroutine until externally resumed
376 if (!t.h_)
377 {
378 note_suspended(h);
379 return std::noop_coroutine();
380 }
381
382 // Transfer ownership into a raw coroutine_handle
383 auto h_new = t.h_.promise().release_ownership();
384 CHECK(h_new, "coroutine handle was not exclusively owned!");
385
386 // If the same AsyncTask is given back to us, resume this coroutine immediately.
387 if (h_new == h)
388 {
389 note_running(h);
390 return h;
391 }
392
393 // Otherwise run a nested coroutine and continue back here once that coroutine completes.
394 h_new.promise().continuation_ = h;
395 note_suspended(h);
396 note_running(h_new);
397 return h_new;
398 }

473 std::coroutine_handle<> await_suspend(std::coroutine_handle<AsyncTask::promise_type> h) noexcept
474 {
475 auto continuation = std::exchange(h.promise().continuation_, nullptr);
476 bool cancelled = h.promise().is_cancel_on_resume();
477 auto eptr = h.promise().get_exception();
478 TRACE_MODULE(ASYNC, "Final suspend of coroutine", h, continuation, cancelled);
479
480 promise_type::destroy_and_track(h);
481
482 TRACE_MODULE(ASYNC, "Destroy is done");
483
484 if (cancelled)
485 {
486 while (continuation)
487 continuation = continuation.promise().destroy_with_continuation();
488 }
489
490 if (continuation)
491 {
492 if (eptr) continuation.promise().set_exception(eptr);
493 promise_type::note_running(continuation);
494 return continuation;
495 }
496 else
497 return std::noop_coroutine();
498 }

500 void await_resume() noexcept {}
501 };

594 AsyncTaskFactory(HandleType h, int count) : handle_(h), count_(count)
595 {
596 [[maybe_unused]] int prior_count = handle_.promise().add_awaiter(count);
597 DEBUG_CHECK_EQUAL(prior_count, 0, "expected handle to be previously unowned!");
598 // if we requested zero references, then we can destroy the handle immediately
599 if (count_ == 0)
600 {
601 AsyncTask::promise_type::destroy_and_track(handle_);
602 handle_ = {};
603 }
604 }
605
607 void release_outstanding() noexcept
608 {
609 // Return the outstanding references. If this reaches zero, destroy the coroutine.
610 if (count_ > 0 && handle_.promise().release_awaiter(count_))
611 {
612 AsyncTask::promise_type::destroy_and_track(handle_);
613 }
614 count_ = 0;
615 handle_ = {};
616 }

618 HandleType handle_;
619 size_t count_;

627template <AsyncTaskAwaitable A>
628auto BasicAsyncTaskPromise::suspend_task_awaitable(std::coroutine_handle<promise_type> h, A& a)
629{
630 using await_return_type = decltype(a.await_suspend(std::declval<AsyncTask>()));
631 if constexpr (std::is_void_v<await_return_type>)
632 {
633 auto t = h.promise().take_ownership();
634 a.await_suspend(std::move(t));
635 promise_type::note_suspended(h);
636 return;
637 }
638 else if constexpr (std::is_same_v<await_return_type, AsyncTask>)
639 {
640 auto t_owning = h.promise().take_ownership();
641 auto t = a.await_suspend(std::move(t_owning));
642 return promise_type::resolve_await_suspend_result(h, std::move(t));
643 }
644 else
645 {
646 static_assert(std::is_same_v<await_return_type, void>,
647 "Unsupported await_suspend() return type: must be void or AsyncTask");
648 }
649}
