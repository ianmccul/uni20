68template <typename Mapping> auto make_iteration_plan_with_offset(Mapping const& mapping)
69{
70 using size_type = typename Mapping::size_type;
71 using index_type = typename Mapping::index_type;
72 static constexpr size_type Rank = Mapping::extents_type::rank();
73
74 static_vector<extent_strides<1>, Rank> raw_plan;
75 static_vector<extent_stride<size_type, index_type>, Rank> plan;
76 index_type offset = 0;
77
78 for (size_type i = 0; i < Rank; ++i)
79 {
80 size_type const extent = mapping.extents().extent(i);
81 index_type stride = mapping.stride(i);
82
83 if (stride < 0)
84 {
85 offset += stride * static_cast<index_type>(extent - 1);
86 stride = -stride;
87 }
88
89 if (extent == 0) continue;
90
91 raw_plan.emplace_back(extent, std::array<index_type, 1>{stride});
92 }
93
94 if (raw_plan.empty()) return std::pair{plan, offset};
95
96 std::sort(raw_plan.begin(), raw_plan.end(),
97 [](auto const& lhs, auto const& rhs) { return std::abs(lhs.strides[0]) > std::abs(rhs.strides[0]); });
98
99 merge_strides_right(raw_plan);
100
101 for (auto const& dim : raw_plan)
102 {
103 plan.emplace_back(static_cast<size_type>(dim.extent), static_cast<index_type>(dim.strides[0]));
104 }
105
106 return std::pair{plan, offset};
107}

115template <typename Mapping, std::size_t N>
116auto make_multi_iteration_plan_with_offset(std::array<Mapping, N> const& mappings)
117{
118 using size_type = typename Mapping::size_type;
119 using index_type = typename Mapping::index_type;
120 static constexpr size_type Rank = Mapping::extents_type::rank();
121
122 static_assert(N >= 1, "At least one mapping is required");
123
124 auto const& base_extents = mappings[0].extents();
125
126 for (std::size_t k = 1; k < N; ++k)
127 {
128 for (size_type i = 0; i < Rank; ++i)
129 {
130 assert(mappings[k].extents().extent(i) == base_extents.extent(i));
131 }
132 }
133
134 static_vector<extent_strides<N>, Rank> raw_plan;
135 std::array<index_type, N> offsets{};
136
137 for (size_type i = 0; i < Rank; ++i)
138 {
139 size_type const extent = base_extents.extent(i);
140 if (extent == 0) continue;
141
142 std::array<index_type, N> strides{};
143 for (std::size_t k = 0; k < N; ++k)
144 {
145 strides[k] = mappings[k].stride(i);
146 }
147
148 if (strides[0] < 0)
149 {
150 for (std::size_t k = 0; k < N; ++k)
151 {
152 offsets[k] += strides[k] * static_cast<index_type>(extent - 1);
153 strides[k] = -strides[k];
154 }
155 }
156
157 raw_plan.emplace_back(extent, strides);
158 }
159
160 if (raw_plan.empty()) return std::pair{raw_plan, offsets};
161
162 std::sort(raw_plan.begin(), raw_plan.end(),
163 [](auto const& lhs, auto const& rhs) { return std::abs(lhs.strides[0]) > std::abs(rhs.strides[0]); });
164
165 merge_strides_right(raw_plan);
166
167 return std::pair{raw_plan, offsets};
