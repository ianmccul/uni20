120template <StridedMdspan First, StridedMdspan... Rest> auto sum_view(First const& first, Rest const&... rest)
121{
122 static_assert((... && (First::rank() == Rest::rank())), "sum_view: rank mismatch");
123 assert(((first.extents() == rest.extents()) && ...) && "sum_view: shape mismatch");
124
125 // 1) Compute merged extents and build the mapping:
126 using CE = common_extents_t<First, Rest...>;
127 CE ext = make_common_extents(first, rest...);
128
129 constexpr std::size_t N = 1 + sizeof...(Rest);
130 constexpr std::size_t R = CE::rank();
131
132 // 2) Pack per-span stride arrays into a [NumSpans][Rank] array:
133 std::array<std::array<typename CE::index_type, R>, N> strides_pack{};
134 {
135 auto fill = [&](auto const& md, std::size_t s) {
136 auto st = md.mapping().strides();
137 for (std::size_t d = 0; d < R; ++d)
138 strides_pack[s][d] = st[d];
139 };
140 fill(first, 0);
141 std::size_t idx = 1;
142 (fill(rest, idx++), ...);
143 }
144
145 // 3) Construct the mdspan mapping
146 using Layout = StridedZipLayout<N>;
147 using Mapping = Layout::template mapping<CE>;
148 auto mapping = Mapping{ext, strides_pack};
149
150 // 4) Pack data_handles and accessors
151 auto handles = std::tuple{first.data_handle(), rest.data_handle()...};
152
153 using accessor_type = SumAccessor<First, Rest...>;
154 accessor_type accessor{first.accessor(), rest.accessor()...};
155
156 // 5) Build resulting mdspan
157 return stdex::mdspan<typename accessor_type::element_type, CE, Layout, SumAccessor<First, Rest...>>{handles, mapping,
158 accessor};
159}

193{
194 // Compute merged extents type & object
195 using CE = common_extents_t<A, B>;
196
197 // Build the new mapping by prepending A onto B’s mapping
198 // Extract how many spans B already has:
199 constexpr size_t M = B::mapping_type::num_spans;
200
201 using NewLayout = StridedZipLayout<M + 1>;
202 using Mapping = NewLayout::template mapping<CE>;
203 Mapping map(a.mapping().strides(), b.mapping());

206 using accessor_type = detail::join_sum_acc_t<A, typename B::accessor_type>;
