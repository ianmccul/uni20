103template <typename TLhs, typename LhsTraits, typename TRhs, typename RhsTraits, typename TOut, typename OutTraits>

105 TensorView<TOut, OutTraits> out)
106{
107 util::require_rank_two(lhs);
108 util::require_rank_two(rhs);
109 util::require_rank_two(out);
110
111 if (lhs.cols() != rhs.rows())
112 {
113 throw std::invalid_argument("matrix dimensions do not agree for multiplication");
114 }
115
116 if (out.rows() != lhs.rows() || out.cols() != rhs.cols())
117 {
118 throw std::invalid_argument("output matrix has incompatible dimensions for multiplication");
119 }
120

124 auto rhs_span = util::const_span(rhs);

126
127 for (index_type i = 0; i < lhs.rows(); ++i)
128 {
129 for (index_type j = 0; j < rhs.cols(); ++j)
130 {
131 value_type value{};
132 for (index_type k = 0; k < lhs.cols(); ++k)

134 value += lhs_span[i, k] * rhs_span[k, j];

231template <typename T, typename Traits> double matrix_one_norm(TensorView<T const, Traits> mat)
232{
233 util::require_rank_two(mat);

286template <typename TA, typename ATraits, typename TB, typename BTraits>
287void solve_linear_system(TensorView<TA, ATraits> A, TensorView<TB, BTraits> B)
288{
289 static_assert(!std::is_const_v<TA>, "Coefficient matrix must be mutable");
290 static_assert(!std::is_const_v<TB>, "Right-hand side matrix must be mutable");
291
292 util::require_rank_two(A);
293 util::require_rank_two(B);
294
295 if (A.rows() != A.cols() || A.rows() != B.rows())
296 {
297 throw std::invalid_argument("solve_linear_system requires square coefficient matrix");
298 }
299
300 index_type const n = A.rows();
301 index_type const nrhs = B.cols();
302
303 auto A_span = util::mutable_span(A);
304 auto B_span = util::mutable_span(B);
305
306 using value_type = std::remove_cv_t<typename TensorView<TB, BTraits>::value_type>;
307
308 for (index_type k = 0; k < n; ++k)
309 {
310 index_type pivot_row = k;
311 double pivot_value = std::abs(A_span[k, k]);
312 for (index_type i = k + 1; i < n; ++i)
313 {
314 double const candidate = std::abs(A_span[i, k]);
315 if (candidate > pivot_value)
316 {
317 pivot_value = candidate;
318 pivot_row = i;
319 }
320 }
321
322 if (pivot_value == 0.0)
323 {
324 throw std::runtime_error("singular matrix in solve_linear_system");
325 }
326
327 if (pivot_row != k)
328 {
329 swap_rows(A, k, pivot_row);
330 swap_rows(B, k, pivot_row);
331 A_span = util::mutable_span(A);
332 B_span = util::mutable_span(B);
333 }
334
335 value_type const pivot = A_span[k, k];
336 for (index_type i = k + 1; i < n; ++i)
337 {
338 value_type const factor = A_span[i, k] / pivot;
339 if (factor == value_type{})
340 {
341 continue;
342 }
343 A_span[i, k] = value_type{};
344 for (index_type j = k + 1; j < n; ++j)
345 {
346 A_span[i, j] -= factor * A_span[k, j];
347 }
348 for (index_type j = 0; j < nrhs; ++j)
349 {
350 B_span[i, j] -= factor * B_span[k, j];
351 }
352 }
353 }
354
355 for (index_type i = n; i-- > 0;)
356 {
357 value_type const pivot = A_span[i, i];
358 for (index_type j = 0; j < nrhs; ++j)
359 {
360 value_type value = B_span[i, j];
361 for (index_type k = i + 1; k < n; ++k)
362 {
363 value -= A_span[i, k] * B_span[k, j];
364 }
365 B_span[i, j] = value / pivot;
366 }
367 }
368}
