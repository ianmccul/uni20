253template <typename A, typename B, async_writer Writer, typename Op>
254requires async_binary_applicable<A, B, Op>
255void async_binary_op(A&& a, B&& b, Writer& out, Op op)
256{
257 auto a_buf = read(std::forward<A>(a));
258 auto b_buf = read(std::forward<B>(b));
259 auto out_buf = out.write();
260
261 schedule([](auto a_, auto b_, auto out_, Op op_) static->AsyncTask {
262 auto ab = co_await all(a_, b_);
263 auto tmp = op_(std::get<0>(ab), std::get<1>(ab));
264 a_.release();
265 b_.release();
266 co_await out_ = std::move(tmp);
267 co_return;
268 }(std::move(a_buf), std::move(b_buf), std::move(out_buf), std::move(op)));
269}

284template <typename U, typename T, typename Op> void async_compound_op(U&& rhs, Async<T>& lhs, Op op)
285{
286 auto rhs_buf = read(std::forward<U>(rhs));
287 auto lhs_in = lhs.read();
288 auto lhs_out = lhs.write();
289
290 schedule([](auto lhs_in_, auto rhs_, WriteBuffer<T> out_, Op op_) static->AsyncTask {
291 auto tmp = co_await all(lhs_in_, rhs_);
292 auto& lhs_val = std::get<0>(tmp);
293 auto& rhs_val = std::get<1>(tmp);
294 T out_val(lhs_val);
295 op_(out_val, rhs_val);
296 lhs_in_.release();
297 rhs_.release();
298 co_await out_ = std::move(out_val);
299 co_return;
300 }(std::move(lhs_in), std::move(rhs_buf), std::move(lhs_out), std::move(op)));
301}
