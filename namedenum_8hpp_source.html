3// A class to represent an enumeration that is iterable and has a string name
4// associated with each item.
5//
6// To use, define a class or struct that has 4 members:
7// Enum - must be an enumeration type; enumeration constants must be contiguous and starting from 0.
8// Default - a constexpr value of type Enum, that is the value of a default-constructed NamedEnumeration
9// StaticName - must be a static constexpr of type const char* that gives a desciption of the enumeration.
10// Names - must be a static constexpr std::array of strings, of exactly the same size as Enum.
11//
12
13// An example:
14// struct MyEnumTraits
15// {
16// enum Enum { Some, Enumeration, Elements };
17// inline static constexpr Enum Default = Enumeration;
18// inline static constexpr const char* StaticName = "the example enumeration";
19// inline static constexpr std::array<const char*, 3> Names = { "some", "enumeration", "elements" };
20// };
21//
22// When constructing a NamedEnumeration from a string, the name is not case sensitive.
23//
