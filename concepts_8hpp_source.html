326template <typename IndexType, std::size_t... StaticExts, typename CharT>
327struct formatter<stdex::extents<IndexType, StaticExts...>, CharT>
328{
329 // No format‐specs, just use the default
330 constexpr auto parse(format_parse_context& ctx) { return ctx.begin(); }
331
332 template <typename FormatContext>
333 constexpr auto format(stdex::extents<IndexType, StaticExts...> const& ex, FormatContext& ctx) const
334 {
335 // write: '[' n0 ',' n1 ',' … ']'
336 auto out = ctx.out();
337 *out++ = '[';
338 constexpr std::size_t R = sizeof...(StaticExts);
339 for (std::size_t d = 0; d < R; ++d)
340 {
341 if (d) *out++ = ',';
342 out = fmt::format_to(out, "{}", ex.extent(d));
343 }
344 *out++ = ']';
345 return out;
346 }
